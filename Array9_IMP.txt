class Solution {
  public:
    int getMinDiff(int arr[], int n, int k) {
        sort(arr,arr+n);
        int res=arr[n-1]-arr[0];
        int minn=arr[0]+k, maxx=arr[n-1]-k;
        for(int i=1; i<n; i++)
        {
            if(arr[i]-k<0) continue;
            
            minn=min(arr[0]+k,arr[i]-k);
            maxx=max(arr[i-1]+k,arr[n-1]-k);
            
            res=min(res,maxx-minn);
        }
        return res;
    }
};
